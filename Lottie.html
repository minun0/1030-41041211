Lottie 動畫

Lottie.js 官網

範例程式碼


<!-- Lottie 容器 -->
<div id="lottieAnimation" style="width: 300px; height: 300px;"></div>

<script>
    // 使用 lottie.loadAnimation 加載動畫
    lottie.loadAnimation({
        container: document.getElementById('lottieAnimation'), // 容器的 ID
        renderer: 'svg', // 渲染類型 (svg, canvas, html)
        loop: true, // 是否循環播放
        autoplay: true, // 是否自動播放
        path: 'https://lottie.host/798222eb-e250-4317-8f17-3943b010e48c/iMo0E4RSgn.json' // Lottie 動畫的 JSON 路徑
    });
</script>
