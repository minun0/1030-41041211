D3.js 官網

範例程式碼


            // 簡單的 D3.js 範例
            const data = [30, 80, 45, 60, 20, 90, 55];
    
            d3.select("body")
                .append("svg")
                .attr("width", 400)
                .attr("height", 100)
                .selectAll("rect")
                .data(data)
                .enter()
                .append("rect")
                .attr("width", (d) => d)
                .attr("height", 20)
                .attr("x", (d, i) => i * 90)
                .attr("y", 0)
                .attr("fill", "steelblue");
