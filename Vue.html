Vue.js 官網

範例程式碼


            const app = Vue.createApp({
                data() {
                    return {
                        count: 0
                    }
                },
                methods: {
                    increment() {
                        this.count++;
                    },
                    decrement() {
                        this.count--;
                    }
                }
            });
            
            app.mount('#app');
